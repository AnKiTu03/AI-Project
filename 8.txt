# Use a base image
FROM nginx:latest
COPY index.html /usr/share/nginx/html/index.html


index.html

<html>
    <body>
        <h1>Hello from Kubernetes!</h1>
    </body>
</html>


docker build -t myapp:latest .

docker tag myapp:latest <your-dockerhub-username>/myapp:latest

docker push <your-dockerhub-username>/myapp:latest


deployment.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: myapp-deployment
spec:
  replicas: 2
  selector:
    matchLabels:
      app: myapp
  template:
    metadata:
      labels:
        app: myapp
    spec:
      containers:
      - name: myapp
        image: <your-dockerhub-username>/myapp:latest
        ports:
        - containerPort: 80
---
apiVersion: v1
kind: Service
metadata:
  name: myapp-service
spec:
  type: LoadBalancer
  selector:
    app: myapp
  ports:
  - protocol: TCP
    port: 80
    targetPort: 80



minikube start
kubectl apply -f deployment.yaml
kubectl get pods
kubectl get services
